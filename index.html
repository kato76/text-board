"; const REMOTE_SEC = 40; const MIC_SEC = 30; // ================================ // 変数 // ================================ let recognition = null; let micTimer = null; let lastMicTime = Date.now(); let micAlive = false; const keepalive = document.getElementById("keepalive"); // ================================ // 初回タップ // ================================ document.getElementById("start").onclick = () => { document.getElementById("start").style.display = "none"; const t = document.getElementById("text"); t.style.display = "flex"; t.innerText = "待機中"; startMic(); }; // ================================ // 表示処理 // ================================ function showText(msg, sec) { const t = document.getElementById("text"); t.innerText = msg; if (msg.length <= 20) t.style.fontSize = "12vw"; else if (msg.length <=40) t.style.fontSize = "10vw"; else if (msg.length <=80) t.style.fontSize = "8vw"; else t.style.fontSize = "6vw"; if (micTimer) clearTimeout(micTimer); micTimer = setTimeout(() => { t.innerText = "待機中"; t.style.fontSize = "6vw"; }, sec * 1000); } // ================================ // スプレッドシート送信 // ================================ function sendToSheet(text) { fetch(POST_URL, { method: "POST", mode: "no-cors", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ text }) }); } // ================================ // マイク開始 // ================================ function startMic() { const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { alert("音声認識に対応していません"); return; } if (recognition) { try { recognition.stop(); } catch(e) {} } recognition = new SpeechRecognition(); recognition.lang = "ja-JP"; recognition.onresult = (event) => { micAlive = true; lastMicTime = Date.now(); const text = event.results[0][0].transcript; showText(text, MIC_SEC); sendToSheet(text); // ★ ここでスプレッドシートへ送信 }; recognition.onend = () => { micAlive = false; setTimeout(startMic, 1000); }; keepalive.volume = 0.001; keepalive.play().catch(()=>{}); recognition.start(); } // ================================ // マイク監視 // ================================ setInterval(() => { if (micAlive && Date.now() - lastMicTime > 60000) { try { recognition.stop();
